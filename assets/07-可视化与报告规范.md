## 排名图代码骨架

```python
import matplotlib.pyplot as plt

def plot_ranking(df, title, x_col, y_col="主体", top_n=15, out_path="fig01.png"):
    sub = df.nlargest(top_n, x_col).sort_values(x_col)
    fig, ax = plt.subplots(figsize=(10, 6))
    ax.barh(sub[y_col], sub[x_col], color="#0F4D92")
    ax.set_xlabel(x_col)
    ax.set_title(title)
    for i, v in enumerate(sub[x_col]):
        ax.text(v, i, f" {v:.2f}", va="center", fontsize=9)
    fig.savefig(out_path, bbox_inches="tight", dpi=300)
    plt.close(fig)
```
