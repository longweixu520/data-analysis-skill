## 质量检查表生成

```python
def check_quality(long_df: pd.DataFrame) -> pd.DataFrame:
    issues = []
    # 缺失
    miss = long_df[long_df["原始值"].isna()]
    for _, row in miss.iterrows():
        issues.append({**row.to_dict(), "问题类型": "结构性空值", "处理建议": "保留"})
    # 计数型小数
    count_indicators = {"人口", "企业数", "学校数"}
    mask = long_df["指标名称"].isin(count_indicators) & (long_df["原始值"] % 1 != 0)
    for _, row in long_df[mask].iterrows():
        issues.append({**row.to_dict(), "问题类型": "单位数小数", "处理建议": "置空"})
    return pd.DataFrame(issues)
```

def normalize_subject(s: str) -> str:
    s = str(s).strip()

## 清洗流水线

```python
def clean_pipeline(raw_long: pd.DataFrame) -> pd.DataFrame:
    df = raw_long.copy()
    df["主体"] = df["主体"].map(normalize_subject)
    df["清洗值"] = pd.to_numeric(df["原始值"], errors="coerce")
    df = apply_structural_fixes(df)      # 错位置空
    df = apply_interpolation(df)         # 时间序列插值
    df = apply_median_fill(df)           # 中位数（带说明）
    return df
```
