## 模式 A：主体 × 年份矩阵

### 方式一：CLI（最快）

```bash
python scripts/parse_matrix.py 原始数据/gdp.xlsx \
  --indicator GDP --category 经济 --unit 亿元 \
  -o output/01_原始标准长表.csv
```

多 sheet 时循环调用，最后 `pd.concat`。

### 方式二：项目内代码

```python
# src/parse/city_year_gdp.py
from pathlib import Path
import sys
sys.path.insert(0, str(Path(__file__).resolve().parents[2] / "scripts"))
from lib.parse_matrix import parse_excel_matrix

def parse(path: Path) -> pd.DataFrame:
    return parse_excel_matrix(
        path,
        sheet_name="GDP",
        subject_col="城市",
        indicator_name="GDP",
        indicator_category="经济",
        unit="亿元",
    )
```

## 合并多指标

```python
from pathlib import Path
import pandas as pd

SHEET_CONFIG = [
    {"sheet": "GDP", "indicator": "GDP", "category": "经济", "unit": "亿元"},
    {"sheet": "人口", "indicator": "常住人口", "category": "社会", "unit": "万人"},
]

def build_long_table(data_dir: Path) -> pd.DataFrame:
    parts = []
    for cfg in SHEET_CONFIG:
        parts.append(parse_one(data_dir / "stats.xlsx", **cfg))
    return pd.concat(parts, ignore_index=True)
```
